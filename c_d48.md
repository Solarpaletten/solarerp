### 4 блока по ТЗ Leanid'а

**Block 1 — Header**: Name, Code (font-mono), Type label, Status badge (Active/Inactive зелёный/красный), breadcrumb `← Customers`

**Block 2 — Form**: General Information (name, display name, code UPPERCASE max 20, type, location, VAT, active, auto-remind), Contact (email, phone, fax), Address (country, city, address, postal code)

**Block 3 — Accounting** (правый столбец): Credit limit, Currency, Payment terms. Bank Account (IBAN, bank name, BLZ, SWIFT). Подготовка для SKR03/DATEV.

**Block 4 — Actions**: Save (blue), Close (gray), Archive/deactivate (amber, soft delete), Delete permanently (red, hard delete с `?hard=true`, блокируется если есть документы)

**Dual mode**: `/clients/new` → POST create → redirect на edit. `/clients/[id]` → GET load → PATCH save.

**Validation**: name required, code required (UPPERCASE, no spaces).

// app/(dashboard)/company/[companyId]/clients/[clientId]/page.tsx
// ═══════════════════════════════════════════════════
// Task 48: Client Detail Editor (production quality)
// ═══════════════════════════════════════════════════
// 4 blocks: Header | Form | Accounting | Actions
// Dual mode: /clients/new (create) | /clients/[id] (edit)
// API: GET + PATCH + DELETE existing endpoints

'use client';

import { useState, useEffect, useCallback } from 'react';
import { useParams, useRouter } from 'next/navigation';

// ─── Types ───────────────────────────────────────

type ClientData = {
  id?: string;
  name: string;
  shortName: string;
  code: string;
  type: string;
  location: string;
  isActive: boolean;
  vatCode: string;
  businessLicenseCode: string;
  residentTaxCode: string;
  email: string;
  phoneNumber: string;
  faxNumber: string;
  contactInfo: string;
  notes: string;
  payWithinDays: string;
  creditLimit: string;
  creditLimitCurrency: string;
  automaticDebtRemind: boolean;
  birthday: string;
  registrationCountryCode: string;
  registrationCity: string;
  registrationAddress: string;
  registrationZipCode: string;
  correspondenceCountryCode: string;
  correspondenceCity: string;
  correspondenceAddress: string;
  correspondenceZipCode: string;
  bankAccount: string;
  bankName: string;
  bankCode: string;
  bankSwiftCode: string;
};

const EMPTY: ClientData = {
  name: '', shortName: '', code: '', type: 'COMPANY', location: 'LOCAL',
  isActive: true, vatCode: '', businessLicenseCode: '', residentTaxCode: '',
  email: '', phoneNumber: '', faxNumber: '', contactInfo: '', notes: '',
  payWithinDays: '', creditLimit: '', creditLimitCurrency: 'EUR',
  automaticDebtRemind: false, birthday: '',
  registrationCountryCode: '', registrationCity: '',
  registrationAddress: '', registrationZipCode: '',
  correspondenceCountryCode: '', correspondenceCity: '',
  correspondenceAddress: '', correspondenceZipCode: '',
  bankAccount: '', bankName: '', bankCode: '', bankSwiftCode: '',
};

const CLIENT_TYPES = [
  { value: 'COMPANY', label: 'Company' },
  { value: 'SOLE_TRADER', label: 'Sole Trader' },
  { value: 'INDIVIDUAL', label: 'Individual' },
  { value: 'GOVERNMENT', label: 'Government' },
  { value: 'NON_PROFIT', label: 'Non-Profit' },
];

const LOCATIONS = [
  { value: 'LOCAL', label: 'Local' },
  { value: 'EU', label: 'EU' },
  { value: 'FOREIGN', label: 'Foreign' },
];

const TYPE_LABELS: Record<string, string> = Object.fromEntries(CLIENT_TYPES.map(t => [t.value, t.label]));

// ─── Helpers ─────────────────────────────────────

const INPUT = "w-full px-3 py-1.5 text-sm border border-gray-200 rounded-md focus:outline-none focus:ring-1 focus:ring-blue-400 focus:border-blue-400 bg-white";
const SELECT = `${INPUT} appearance-none`;

function Field({ label, children, required }: { label: string; children: React.ReactNode; required?: boolean }) {
  return (
    <div>
      <label className="block text-xs font-medium text-gray-500 mb-1">
        {label}{required && <span className="text-red-500 ml-0.5">*</span>}
      </label>
      {children}
    </div>
  );
}

function Section({ title, color, children }: { title: string; color?: string; children: React.ReactNode }) {
  return (
    <div className="bg-white rounded-lg border border-gray-200 p-5 space-y-4">
      <h3 className={`text-sm font-semibold ${color || 'text-gray-700'} border-b border-gray-100 pb-2`}>{title}</h3>
      {children}
    </div>
  );
}

// ─── Main Component ──────────────────────────────

export default function ClientEditorPage() {
  const params = useParams();
  const router = useRouter();
  const companyId = params.companyId as string;
  const clientId = params.clientId as string;
  const isNew = clientId === 'new';
  const base = `/company/${companyId}`;

  const [data, setData] = useState<ClientData>({ ...EMPTY });
  const [loading, setLoading] = useState(!isNew);
  const [saving, setSaving] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [saveMsg, setSaveMsg] = useState<string | null>(null);
  const [dirty, setDirty] = useState(false);

  // ── Load ───────────────────────────────────────

  const loadClient = useCallback(async () => {
    if (isNew) return;
    setLoading(true);
    try {
      const res = await fetch(`/api/company/${companyId}/clients/${clientId}`, { cache: 'no-store' });
      if (!res.ok) throw new Error('Client not found');
      const json = await res.json();
      const d = json.data;
      setData({
        id: d.id,
        name: d.name || '', shortName: d.shortName || '', code: d.code || '',
        type: d.type || 'COMPANY', location: d.location || 'LOCAL',
        isActive: d.isActive !== false,
        vatCode: d.vatCode || '', businessLicenseCode: d.businessLicenseCode || '',
        residentTaxCode: d.residentTaxCode || '',
        email: d.email || '', phoneNumber: d.phoneNumber || '', faxNumber: d.faxNumber || '',
        contactInfo: d.contactInfo || '', notes: d.notes || '',
        payWithinDays: d.payWithinDays != null ? String(d.payWithinDays) : '',
        creditLimit: d.creditLimit != null ? String(d.creditLimit) : '',
        creditLimitCurrency: d.creditLimitCurrency || 'EUR',
        automaticDebtRemind: d.automaticDebtRemind || false,
        birthday: d.birthday || '',
        registrationCountryCode: d.registrationCountryCode || '',
        registrationCity: d.registrationCity || '',
        registrationAddress: d.registrationAddress || '',
        registrationZipCode: d.registrationZipCode || '',
        correspondenceCountryCode: d.correspondenceCountryCode || '',
        correspondenceCity: d.correspondenceCity || '',
        correspondenceAddress: d.correspondenceAddress || '',
        correspondenceZipCode: d.correspondenceZipCode || '',
        bankAccount: d.bankAccount || '', bankName: d.bankName || '',
        bankCode: d.bankCode || '', bankSwiftCode: d.bankSwiftCode || '',
      });
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to load');
    } finally {
      setLoading(false);
    }
  }, [companyId, clientId, isNew]);

  useEffect(() => { loadClient(); }, [loadClient]);

  // ── Update field ───────────────────────────────

  const set = (key: keyof ClientData, value: string | boolean) => {
    setData(prev => ({ ...prev, [key]: value }));
    setDirty(true);
  };

  const setCode = (val: string) => set('code', val.toUpperCase().replace(/\s/g, '').slice(0, 20));

  // ── Validate ───────────────────────────────────

  const validate = (): string | null => {
    if (!data.name.trim()) return 'Name is required';
    if (!data.code.trim()) return 'Code is required';
    return null;
  };

  // ── Save ───────────────────────────────────────

  const handleSave = async () => {
    const err = validate();
    if (err) { setError(err); return; }

    setSaving(true); setError(null); setSaveMsg(null);
    try {
      const url = isNew ? `/api/company/${companyId}/clients` : `/api/company/${companyId}/clients/${clientId}`;
      const method = isNew ? 'POST' : 'PATCH';

      const payload = {
        ...data,
        payWithinDays: data.payWithinDays ? Number(data.payWithinDays) : null,
        creditLimit: data.creditLimit ? Number(data.creditLimit) : null,
      };

      const res = await fetch(url, { method, headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
      if (!res.ok) { const e = await res.json(); throw new Error(e.error || 'Save failed'); }

      const json = await res.json();
      setDirty(false);
      setSaveMsg('Saved');
      setTimeout(() => setSaveMsg(null), 2500);

      if (isNew) {
        router.replace(`${base}/clients/${json.data.id}`);
      } else {
        loadClient();
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Save failed');
    } finally {
      setSaving(false);
    }
  };

  // ── Archive (soft delete) ──────────────────────

  const handleArchive = async () => {
    if (!confirm('Deactivate this client?')) return;
    try {
      const res = await fetch(`/api/company/${companyId}/clients/${clientId}`, { method: 'DELETE' });
      if (!res.ok) { const e = await res.json(); throw new Error(e.error || 'Archive failed'); }
      router.push(`${base}/clients`);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Archive failed');
    }
  };

  // ── Hard Delete ────────────────────────────────

  const handleDelete = async () => {
    if (!confirm('Permanently delete this client? This is only possible if no documents are linked.')) return;
    try {
      const res = await fetch(`/api/company/${companyId}/clients/${clientId}?hard=true`, { method: 'DELETE' });
      if (!res.ok) { const e = await res.json(); throw new Error(e.error || 'Delete failed'); }
      router.push(`${base}/clients`);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Delete failed');
    }
  };

  // ── Loading state ──────────────────────────────

  if (loading) {
    return (
      <div className="flex items-center justify-center h-full py-20">
        <div className="flex items-center gap-2 text-gray-400 text-sm">
          <svg className="animate-spin h-4 w-4" viewBox="0 0 24 24">
            <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4" fill="none" />
            <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z" />
          </svg>
          Loading client...
        </div>
      </div>
    );
  }

  // ── Render ─────────────────────────────────────

  return (
    <div className="max-w-4xl mx-auto p-6 space-y-6">

      {/* ══ BLOCK 1: Header ══ */}
      <div className="flex items-center justify-between">
        <div>
          <div className="flex items-center gap-2 mb-1">
            <a href={`${base}/clients`} className="text-xs text-gray-400 hover:text-gray-600">&larr; Customers</a>
          </div>
          <h1 className="text-xl font-bold text-gray-900">
            {isNew ? 'New Client' : data.name || 'Client'}
          </h1>
          {!isNew && data.code && (
            <p className="text-sm text-gray-500 mt-0.5">
              Code: <span className="font-mono font-medium">{data.code}</span>
              {' \u00b7 '}
              Type: {TYPE_LABELS[data.type] || data.type}
            </p>
          )}
        </div>
        <div className="flex items-center gap-2">
          {!isNew && (
            <span className={`px-2.5 py-1 text-[10px] font-bold uppercase tracking-wider rounded border ${
              data.isActive
                ? 'bg-emerald-50 text-emerald-700 border-emerald-200'
                : 'bg-red-50 text-red-600 border-red-200'
            }`}>
              {data.isActive ? 'Active' : 'Inactive'}
            </span>
          )}
        </div>
      </div>

      {/* Error / Success banners */}
      {error && (
        <div className="bg-red-50 border border-red-200 rounded-lg px-4 py-2.5 text-sm text-red-700 flex items-center justify-between">
          <span>{error}</span>
          <button onClick={() => setError(null)} className="text-red-400 hover:text-red-600 text-xs ml-4">&times;</button>
        </div>
      )}
      {saveMsg && (
        <div className="bg-emerald-50 border border-emerald-200 rounded-lg px-4 py-2 text-sm text-emerald-700">
          {saveMsg}
        </div>
      )}

      {/* ══ BLOCK 2: General Information Form ══ */}
      <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
        <div className="lg:col-span-2 space-y-4">
          <Section title="General Information">
            <div className="grid grid-cols-2 gap-4">
              <Field label="Name" required><input className={INPUT} value={data.name} onChange={e => set('name', e.target.value)} /></Field>
              <Field label="Display name"><input className={INPUT} value={data.shortName} onChange={e => set('shortName', e.target.value)} /></Field>
            </div>
            <div className="grid grid-cols-3 gap-4">
              <Field label="Code" required><input className={INPUT} value={data.code} onChange={e => setCode(e.target.value)} placeholder="APPLE" /></Field>
              <Field label="Type"><select className={SELECT} value={data.type} onChange={e => set('type', e.target.value)}>{CLIENT_TYPES.map(t => <option key={t.value} value={t.value}>{t.label}</option>)}</select></Field>
              <Field label="Location"><select className={SELECT} value={data.location} onChange={e => set('location', e.target.value)}>{LOCATIONS.map(l => <option key={l.value} value={l.value}>{l.label}</option>)}</select></Field>
            </div>
            <Field label="VAT Number"><input className={INPUT} value={data.vatCode} onChange={e => set('vatCode', e.target.value)} placeholder="DE123456789" /></Field>
            <div className="flex items-center gap-6">
              <label className="flex items-center gap-2 text-sm"><input type="checkbox" checked={data.isActive} onChange={e => set('isActive', e.target.checked)} className="rounded border-gray-300" /> Active</label>
              <label className="flex items-center gap-2 text-sm"><input type="checkbox" checked={data.automaticDebtRemind} onChange={e => set('automaticDebtRemind', e.target.checked)} className="rounded border-gray-300" /> Auto debt reminder</label>
            </div>
          </Section>

          <Section title="Contact">
            <div className="grid grid-cols-3 gap-4">
              <Field label="Email"><input className={INPUT} type="email" value={data.email} onChange={e => set('email', e.target.value)} /></Field>
              <Field label="Phone"><input className={INPUT} value={data.phoneNumber} onChange={e => set('phoneNumber', e.target.value)} /></Field>
              <Field label="Fax"><input className={INPUT} value={data.faxNumber} onChange={e => set('faxNumber', e.target.value)} /></Field>
            </div>
          </Section>

          <Section title="Address">
            <div className="grid grid-cols-2 gap-4">
              <Field label="Country"><input className={INPUT} value={data.registrationCountryCode} onChange={e => set('registrationCountryCode', e.target.value)} placeholder="Germany" /></Field>
              <Field label="City"><input className={INPUT} value={data.registrationCity} onChange={e => set('registrationCity', e.target.value)} /></Field>
            </div>
            <Field label="Address"><input className={INPUT} value={data.registrationAddress} onChange={e => set('registrationAddress', e.target.value)} /></Field>
            <Field label="Postal code"><input className={INPUT} value={data.registrationZipCode} onChange={e => set('registrationZipCode', e.target.value)} /></Field>
          </Section>
        </div>

        {/* ══ BLOCK 3: Accounting (right column) ══ */}
        <div className="space-y-4">
          <Section title="Accounting" color="text-blue-700">
            <Field label="Credit limit"><input className={INPUT} type="number" min="0" step="0.01" value={data.creditLimit} onChange={e => set('creditLimit', e.target.value)} /></Field>
            <Field label="Currency"><input className={INPUT} value={data.creditLimitCurrency} onChange={e => set('creditLimitCurrency', e.target.value)} placeholder="EUR" /></Field>
            <Field label="Payment terms (days)"><input className={INPUT} type="number" min="0" value={data.payWithinDays} onChange={e => set('payWithinDays', e.target.value)} /></Field>
          </Section>

          <Section title="Bank Account">
            <Field label="IBAN"><input className={INPUT} value={data.bankAccount} onChange={e => set('bankAccount', e.target.value)} /></Field>
            <Field label="Bank name"><input className={INPUT} value={data.bankName} onChange={e => set('bankName', e.target.value)} /></Field>
            <Field label="BLZ"><input className={INPUT} value={data.bankCode} onChange={e => set('bankCode', e.target.value)} /></Field>
            <Field label="SWIFT/BIC"><input className={INPUT} value={data.bankSwiftCode} onChange={e => set('bankSwiftCode', e.target.value)} /></Field>
          </Section>
        </div>
      </div>

      {/* ══ BLOCK 4: Actions ══ */}
      <div className="flex items-center justify-between pt-2 border-t border-gray-200">
        <div>
          {!isNew && (
            <div className="flex items-center gap-2">
              <button onClick={handleArchive}
                className="px-3 py-1.5 text-xs font-medium text-amber-700 bg-amber-50 border border-amber-200 rounded-md hover:bg-amber-100 transition-colors">
                Archive (deactivate)
              </button>
              <button onClick={handleDelete}
                className="px-3 py-1.5 text-xs font-medium text-red-600 bg-red-50 border border-red-200 rounded-md hover:bg-red-100 transition-colors">
                Delete permanently
              </button>
            </div>
          )}
        </div>
        <div className="flex items-center gap-2">
          <button onClick={() => router.push(`${base}/clients`)}
            className="px-4 py-1.5 text-xs font-medium text-gray-600 bg-white border border-gray-200 rounded-md hover:bg-gray-50 transition-colors">
            Close
          </button>
          <button onClick={handleSave} disabled={saving || (!dirty && !isNew)}
            className="px-4 py-1.5 text-xs font-semibold text-white bg-blue-600 rounded-md hover:bg-blue-700 disabled:bg-blue-300 disabled:cursor-not-allowed transition-colors">
            {saving ? 'Saving...' : 'Save'}
          </button>
        </div>
      </div>
    </div>
  );
}
